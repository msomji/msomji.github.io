Using Constructor Methods as Prototypes in JS and other cheats!

February 13, 2014

Prototypes can get very confusing very fast but to keep things simple, they are basically objects that point to other objects.

Below is an example of a way to pass prototypes through a constructor method

function Vehicle (year,color) { this.year = year; this.color = color; } function Contour(year,color) { this.make = "Ford"; this.model = "Contour"; Vehicle.call(this, year,color); } // create object from prototype var white99contour = new Contour(1999,'White'); console.log(Object.getPrototypeOf(white99contour)) // => Contour {} console.log(white99contour); // => Contour {make: "Ford", model: "Contour", year: 1999, color: "White"}

Below we pass a prototype using Object.create

var dog = { genus: "canis", species: "lupus" } // create basic object from another var puppy = Object.create(dog); console.log(puppy); // => Object {genus: "canis", species: "lupus"} console.log(Object.getPrototypeOf(puppy)) // => Object {genus: "canis", species: "lupus"} // Not super helpful since it the object is being expressed and it is essentially a clone. // But you can test it this way... console.log(Object.getPrototypeOf(puppy) == dog) // => true // Object#create with more properties within the assignment. var tanya = Object.create(dog,{age: {value: 8}, name: {value: "Tanya"}, breed: {value: "Golden-Retriever"}}); console.log(tanya); // => Object {age: 8, name: "Tanya", breed: "Golden-Retriever", genus: "canis", species: "lupus"} console.log(Object.getPrototypeOf(tanya) == dog) // => true // Object#create but add more properties after var sophie = Object.create(dog); sophie.age = 6; sophie.name = "Sophia"; sophie.breed = "Golden-Doodle"; console.log(sophie); // => Object {age: 6, name: "Sophia", breed: "Golden-Doodle", genus: "canis", species: "lupus"} console.log(Object.getPrototypeOf(sophie) == dog) // => true
